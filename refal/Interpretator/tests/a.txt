program a;
    var a,i,j:integer; q:boolean;
begin
begin
a:=-2+1;
writeln(a);
q:=(-1)<2;
repeat
case 0 of 1: begin 
                 writeln(a*2, -1-3); 
                 case (2    mod 1) of 
                   a : writeln(a);
                    3 : writeln(0-a);
                 endcase
              end;
endcase;
for i:=1 to 10 do
for j:=1 to 10 do
  writeln((i-1)*10+j-1);
repeat
q:=not q	
until (not q);
while false do while false do i:=-45 mod 10;
if not q
  then
    for i:=1 to 3 do writeln(i*(i+1));
  else writeln(false);
while (not q = q) do writeln(1);
while (q = false) do q:=true;
begin
q:=not q;
end;
until false;
end;
end.
